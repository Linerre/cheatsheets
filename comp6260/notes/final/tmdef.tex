    \item[$p$] next state, \(p \in Q\)
    \item[$Y$] \(\in \Gamma\), symbol in tape cell, replacing whatever was there
    \item[$D$] \emph{direction}, \(L\leftarrow\), \(R\rightarrow\), \(S\)tay/\(H\)alt

    \end{enumerate}
