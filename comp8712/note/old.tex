  \begin{itemize}
  \item LR($k$) still parses from left to right
  \item Parse tree is drawn from bottom up according to the parse table (as if parsing in reverse order and it is rightmost)
  \item Reduce: If found a matching production on the stack: $A\to \alpha\beta Z$, replace (pop) $\alpha\beta Z$ (RHS) with $A$ (push LHS)
  \item Reduce($k$) Pop stack as many times as the \# of symbols on RHS of rule $k$
  \item \mb{s}$n$: \mr{shift} into state $n$; \mb{g}$n$ \mr{goto} state $n$; (edges of DFA)
  \item \mb{r}$k$: \mr{reduce} by rule $k$; \mb{a}: accept; \$: EOF
  \end{itemize}

  \begin{itemize}
  \item For each edge $I\overset{X}{\to}J$ where $X$ is a \mb{terminal}, put action \mr{shift $J$} at position $(I, X)$ of the table
  \item If $X$ is a \mb{non-terminal}, put \mr{goto $J$} at position $(I,X)$
  \item For each state $I$ containing an item $S'\to S.$, put \mb{accept} at $(I,\$)$. For example, $(I=S_4,\$)$
  \item For any state containing $A\to\gamma.$ (production $n$ with dot at the end), put \mr{reduce $n$} at $(I,Y)$ for every token $Y$.
    For example, $I=S_2$ matches $P_2.$, so for row of $S_2$, each token has $r2$. For $I=S_4$, we do not put $s0$ though, it's special
  \end{itemize}
